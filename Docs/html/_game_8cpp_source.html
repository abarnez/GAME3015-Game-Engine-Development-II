196void Game::UpdateMainPassCB(const GameTimer& gt)
197{
198 XMMATRIX view = mCamera.GetView();
199 XMMATRIX proj = mCamera.GetProj();
200
201 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
202 XMMATRIX invView = XMMatrixInverse(&XMMatrixDeterminant(view), view);
203 XMMATRIX invProj = XMMatrixInverse(&XMMatrixDeterminant(proj), proj);
204 XMMATRIX invViewProj = XMMatrixInverse(&XMMatrixDeterminant(viewProj), viewProj);
205
206 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
207 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
208 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
209 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
210 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
211 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
212 mMainPassCB.EyePosW = mCamera.GetPosition3f();
213 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);
214 mMainPassCB.InvRenderTargetSize = XMFLOAT2(1.0f / mClientWidth, 1.0f / mClientHeight);
215 mMainPassCB.NearZ = 1.0f;
216 mMainPassCB.FarZ = 1000.0f;
217 mMainPassCB.TotalTime = gt.TotalTime();
218 mMainPassCB.DeltaTime = gt.DeltaTime();
219 mMainPassCB.AmbientLight = { 1.0f, 1.0f, 1.0f, 1.0f };
220 mMainPassCB.Lights[0].Direction = { 0.57735f, -0.57735f, 0.57735f };
221 mMainPassCB.Lights[0].Strength = { 1.0f, 1.0f, 0.0f };

311void Game::BuildRootSignature()
312{
313 CD3DX12_DESCRIPTOR_RANGE texTable;
314 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
315
316 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
317 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
318 slotRootParameter[1].InitAsConstantBufferView(0);
319 slotRootParameter[2].InitAsConstantBufferView(1);
320 slotRootParameter[3].InitAsConstantBufferView(2);
321
322
323 auto staticSamplers = GetStaticSamplers();
324
325 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
326 (UINT)staticSamplers.size(), staticSamplers.data(), //6 samplers!
327 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
328
329 ComPtr<ID3DBlob> serializedRootSig = nullptr;
330 ComPtr<ID3DBlob> errorBlob = nullptr;
331 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
332 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
333
334 if (errorBlob != nullptr)
335 {
336 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
337 }
338 ThrowIfFailed(hr);
339
340 ThrowIfFailed(md3dDevice->CreateRootSignature(
341 0,
342 serializedRootSig->GetBufferPointer(),
343 serializedRootSig->GetBufferSize(),
344 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
345}

447
448 std::vector<std::uint16_t> indices = box.GetIndices16();
