196void Game::UpdateMainPassCB(const GameTimer& gt)
197{
198 XMMATRIX view = mCamera.GetView();
199 XMMATRIX proj = mCamera.GetProj();
200
201 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
202 XMMATRIX invView = XMMatrixInverse(&XMMatrixDeterminant(view), view);
203 XMMATRIX invProj = XMMatrixInverse(&XMMatrixDeterminant(proj), proj);
204 XMMATRIX invViewProj = XMMatrixInverse(&XMMatrixDeterminant(viewProj), viewProj);
205
206 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
207 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
208 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
209 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
210 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
211 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
212 mMainPassCB.EyePosW = mCamera.GetPosition3f();
213 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);
214 mMainPassCB.InvRenderTargetSize = XMFLOAT2(1.0f / mClientWidth, 1.0f / mClientHeight);
215 mMainPassCB.NearZ = 1.0f;
216 mMainPassCB.FarZ = 1000.0f;
217 mMainPassCB.TotalTime = gt.TotalTime();
218 mMainPassCB.DeltaTime = gt.DeltaTime();

258{

261

263 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
264 slotRootParameter[1].InitAsConstantBufferView(0);
265 slotRootParameter[2].InitAsConstantBufferView(1);
266 slotRootParameter[3].InitAsConstantBufferView(2);
267
268 auto staticSamplers = GetStaticSamplers();
269
270 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,

274 ComPtr<ID3DBlob> serializedRootSig = nullptr;

277 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());

363 std::vector<std::uint16_t> indices = box.GetIndices16();

375 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
376
377 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
378 mCommandList.Get(), vertices.data(), vbByteSize, geo->VertexBufferUploader);
379
380 geo->IndexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
381 mCommandList.Get(), indices.data(), ibByteSize, geo->IndexBufferUploader);
382
383 geo->VertexByteStride = sizeof(Vertex);
