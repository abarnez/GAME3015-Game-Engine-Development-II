17bool Game::Initialize()
18{
19 if (!D3DApp::Initialize())
20 return false;
21
22
23 mCamera.SetPosition(0, 18, 0);
24 mCamera.Pitch(3.14 / 2);
25
26 // Reset the command list to prep for initialization commands.
27 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));
28
29 // Get the increment size of a descriptor in this heap type. This is hardware specific,
30 // so we have to query this information.
31 mCbvSrvDescriptorSize = md3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
32
33 LoadTextures();
34 BuildRootSignature();
35 BuildDescriptorHeaps();
36 BuildShadersAndInputLayout();
37 BuildShapeGeometry();
38 BuildMaterials();
39 BuildRenderItems();
40 BuildFrameResources();
41 BuildPSOs();
42
43 // Execute the initialization commands.
44 ThrowIfFailed(mCommandList->Close());
45 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
46 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
47
48 // Wait until initialization is complete.
49 FlushCommandQueue();
50
51 return true;
52}

60void Game::Update(const GameTimer& gt)
61{
62 OnKeyboardInput(gt);
63 mWorld.update(gt);
64 //UpdateCamera(gt);
65
66 // Cycle through the circular frame resource array.
67 mCurrFrameResourceIndex = (mCurrFrameResourceIndex + 1) % gNumFrameResources;
68 mCurrFrameResource = mFrameResources[mCurrFrameResourceIndex].get();
69
70 // Has the GPU finished processing the commands of the current frame resource?
71 // If not, wait until the GPU has completed commands up to this fence point.
72 if (mCurrFrameResource->Fence != 0 && mFence->GetCompletedValue() < mCurrFrameResource->Fence)
73 {
74 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
75 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrFrameResource->Fence, eventHandle));
76 WaitForSingleObject(eventHandle, INFINITE);
77 CloseHandle(eventHandle);
78 }
79
80 AnimateMaterials(gt);
81 UpdateObjectCBs(gt);
82 UpdateMaterialCBs(gt);
83 UpdateMainPassCB(gt);

86void Game::Draw(const GameTimer& gt)
87{
88 auto cmdListAlloc = mCurrFrameResource->CmdListAlloc;
89
90 // Reuse the memory associated with command recording.
91 // We can only reset when the associated command lists have finished execution on the GPU.
92 ThrowIfFailed(cmdListAlloc->Reset());
93
94 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
95 // Reusing the command list reuses memory.
96 ThrowIfFailed(mCommandList->Reset(cmdListAlloc.Get(), mOpaquePSO.Get()));
97
98 mCommandList->RSSetViewports(1, &mScreenViewport);
99 mCommandList->RSSetScissorRects(1, &mScissorRect);
100
101 // Indicate a state transition on the resource usage.
102 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
103 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
104
105 // Clear the back buffer and depth buffer.
106 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
107 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
108
109 // Specify the buffers we are going to render to.
110 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
111
112 ID3D12DescriptorHeap* descriptorHeaps[] = { mSrvDescriptorHeap.Get() };
113 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
114
115 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
116
117 auto passCB = mCurrFrameResource->PassCB->Resource();
118 mCommandList->SetGraphicsRootConstantBufferView(2, passCB->GetGPUVirtualAddress());
119
120 mWorld.draw();
121 DrawRenderItems(mCommandList.Get(), mOpaqueRitems);
122
123 // Indicate a state transition on the resource usage.
124 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
125 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
126
127 // Done recording commands.
128 ThrowIfFailed(mCommandList->Close());
129
130 // Add the command list to the queue for execution.
131 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
132 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
133
134 // Swap the back and front buffers
135 ThrowIfFailed(mSwapChain->Present(0, 0));
136 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
137
138 // Advance the fence value to mark commands up to this fence point.
139 mCurrFrameResource->Fence = ++mCurrentFence;
140
141 // Add an instruction to the command queue to set a new fence point.
142 // Because we are on the GPU timeline, the new fence point won't be
143 // set until the GPU finishes processing all the commands prior to this Signal().
144 mCommandQueue->Signal(mFence.Get(), mCurrentFence);
145}

173void Game::UpdateObjectCBs(const GameTimer& gt)
174{
175 auto currObjectCB = mCurrFrameResource->ObjectCB.get();
176 for (auto& e : mAllRitems)
177 {
178 // Only update the cbuffer data if the constants have changed.
179 // This needs to be tracked per frame resource.
180 if (e->NumFramesDirty > 0)
181 {
182 XMMATRIX world = XMLoadFloat4x4(&e->World);
183 XMMATRIX texTransform = XMLoadFloat4x4(&e->TexTransform);
184
185 ObjectConstants objConstants;
186 XMStoreFloat4x4(&objConstants.World, XMMatrixTranspose(world));
187 XMStoreFloat4x4(&objConstants.TexTransform, XMMatrixTranspose(texTransform));
188
189 currObjectCB->CopyData(e->ObjCBIndex, objConstants);
190
191 // Next FrameResource need to be updated too.
192 e->NumFramesDirty--;
193 }
194 }

223void Game::UpdateMainPassCB(const GameTimer& gt)
224{
225 XMMATRIX view = mCamera.GetView();
226 XMMATRIX proj = mCamera.GetProj();
227
228 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
229 XMMATRIX invView = XMMatrixInverse(&XMMatrixDeterminant(view), view);
230 XMMATRIX invProj = XMMatrixInverse(&XMMatrixDeterminant(proj), proj);
231 XMMATRIX invViewProj = XMMatrixInverse(&XMMatrixDeterminant(viewProj), viewProj);
232
233 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
234 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
235 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
236 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
237 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
238 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
239 mMainPassCB.EyePosW = mCamera.GetPosition3f();
240 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);
241 mMainPassCB.InvRenderTargetSize = XMFLOAT2(1.0f / mClientWidth, 1.0f / mClientHeight);
242 mMainPassCB.NearZ = 1.0f;
243 mMainPassCB.FarZ = 1000.0f;
244 mMainPassCB.TotalTime = gt.TotalTime();
245 mMainPassCB.DeltaTime = gt.DeltaTime();
246 mMainPassCB.AmbientLight = { 1.0f, 1.0f, 1.0f, 1.0f };

287void Game::BuildRootSignature()
288{
289 CD3DX12_DESCRIPTOR_RANGE texTable;
290 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
291
292 // Root parameter can be a table, root descriptor or root constants.
293 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
294
295 // Perfomance TIP: Order from most frequent to least frequent.
296 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
297 slotRootParameter[1].InitAsConstantBufferView(0);
298 slotRootParameter[2].InitAsConstantBufferView(1);
299 slotRootParameter[3].InitAsConstantBufferView(2);
300
301 auto staticSamplers = GetStaticSamplers();
302
303 // A root signature is an array of root parameters.
304 //The Init function of the CD3DX12_ROOT_SIGNATURE_DESC class has two parameters that allow you to
305 //define an array of so - called static samplers your application can use.
306 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
307 (UINT)staticSamplers.size(), staticSamplers.data(), //6 samplers!
308 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
309
310 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
311 ComPtr<ID3DBlob> serializedRootSig = nullptr;
312 ComPtr<ID3DBlob> errorBlob = nullptr;
313 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
314 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
315
316 if (errorBlob != nullptr)
317 {
318 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
319 }
320 ThrowIfFailed(hr);
321
322 ThrowIfFailed(md3dDevice->CreateRootSignature(
323 0,
324 serializedRootSig->GetBufferPointer(),
325 serializedRootSig->GetBufferSize(),
326 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
327}

391void Game::BuildShapeGeometry()
392{
393 GeometryGenerator geoGen;
394 GeometryGenerator::MeshData box = geoGen.CreateBox(20, 0.2, 20, 1);
395 SubmeshGeometry boxSubmesh;
396 boxSubmesh.IndexCount = (UINT)box.Indices32.size();
397 boxSubmesh.StartIndexLocation = 0;
398 boxSubmesh.BaseVertexLocation = 0;
399
400 std::vector<Vertex> vertices(box.Vertices.size());
401
402 for (size_t i = 0; i < box.Vertices.size(); ++i)
403 {
404 vertices[i].Pos = box.Vertices[i].Position;
405 vertices[i].Normal = box.Vertices[i].Normal;
406 vertices[i].TexC = box.Vertices[i].TexC;
407 }
408
409 std::vector<std::uint16_t> indices = box.GetIndices16();
410
411 const UINT vbByteSize = (UINT)vertices.size() * sizeof(Vertex);
412 const UINT ibByteSize = (UINT)indices.size() * sizeof(std::uint16_t);
413
414 auto geo = std::make_unique<MeshGeometry>();
415 geo->Name = "boxGeo";
416
417 ThrowIfFailed(D3DCreateBlob(vbByteSize, &geo->VertexBufferCPU));
418 CopyMemory(geo->VertexBufferCPU->GetBufferPointer(), vertices.data(), vbByteSize);
419
420 ThrowIfFailed(D3DCreateBlob(ibByteSize, &geo->IndexBufferCPU));
421 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
422
423 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
424 mCommandList.Get(), vertices.data(), vbByteSize, geo->VertexBufferUploader);
425
426 geo->IndexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
427 mCommandList.Get(), indices.data(), ibByteSize, geo->IndexBufferUploader);
428
429 geo->VertexByteStride = sizeof(Vertex);
430 geo->VertexBufferByteSize = vbByteSize;
431 geo->IndexFormat = DXGI_FORMAT_R16_UINT;
432 geo->IndexBufferByteSize = ibByteSize;
433
434 geo->DrawArgs["box"] = boxSubmesh;
435
436 mGeometries[geo->Name] = std::move(geo);
437}
